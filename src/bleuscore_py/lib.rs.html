4use pyo3::prelude::*;

7#[pyfunction]
8fn tokenizer_regex(line: &str) -> PyResult<Vec<String>> {
9    let tokenizer_regex = TokenizerRegex::new();
10    let res = tokenizer_regex.tokenize(line);
11    Ok(res)
12}
13
14#[pyfunction]
15fn tokenizer_13a(line: &str) -> PyResult<Vec<String>> {
16    let tokenizer_13a_regex = Tokenizer13a::new();
17    let res = tokenizer_13a_regex.tokenize(line);
18    Ok(res)
19}
20
21#[pyfunction]
22#[pyo3(signature = (references, predictions, max_order=4, smooth=false))]
23fn compute(
24    references: Vec<Vec<String>>,
25    predictions: Vec<String>,
26    max_order: usize,
27    smooth: bool,
28) -> PyResult<PyObject> {
29    let bleu = compute_score(&references, &predictions, max_order, smooth);
30    Python::with_gil(|py| {
31        let bleu_dict = PyDict::new(py);
32        bleu_dict.set_item("bleu", bleu.bleu)?;
33        bleu_dict.set_item("precisions", bleu.precisions)?;
34        bleu_dict.set_item("brevity_penalty", bleu.brevity_penalty)?;
35        bleu_dict.set_item("length_ratio", bleu.length_ratio)?;
36        bleu_dict.set_item("translation_length", bleu.translation_length)?;
37        bleu_dict.set_item("reference_length", bleu.reference_length)?;
38        Ok(bleu_dict.into())
39    })
40}
