8/// The BLEU score data struct
9#[derive(Debug, Default)]
10pub struct BleuScore {
11    pub bleu: f64,
12    pub precisions: Vec<f64>,
13    pub brevity_penalty: f64,
14    pub length_ratio: f64,
15    pub translation_length: usize,
16    pub reference_length: usize,
17}

19/// Used to save the statistic result for every prediction-references pair
20#[derive(Debug, Eq, PartialEq)]
21struct Stat(usize, usize, Vec<usize>, Vec<usize>);
22
23/// aggregate prediction-references pairs' statistic result into final result
24fn agg_stat(s1: Stat, s2: Stat) -> Stat {
25    Stat(
26        s1.0 + s2.0,
27        s1.1 + s2.1,
28        add_vectors(s1.2, s2.2),
29        add_vectors(s1.3, s2.3),
30    )
31}

33/// compute the BLEU score with `Tokenizer13a` as default tokenizer.
34/// The implementation is based on [huggingface/nmt](https://github.com/huggingface/evaluate/blob/main/metrics/bleu/bleu.py)
35pub fn compute_score(
36    references: &[Vec<String>],
37    predictions: &[String],
38    max_order: usize,
39    smooth: bool,
40) -> BleuScore {
41    let tokenizer = Tokenizer13a::new();
42
43    // stat calculation
44    let stat_result = references
45        .into_par_iter()
46        .zip(predictions.into_par_iter())
47        .map(|(references, translation)| {
48            let translation_tokens = tokenizer.tokenize(translation);
49            let references_tokens: Vec<Vec<String>> =
50                references.iter().map(|x| tokenizer.tokenize(x)).collect();
51
52            let reference_length = references_tokens.iter().map(|x| x.len()).min().unwrap();
53            let translation_length = translation_tokens.len();
54
55            let mut possible_matches_by_order = vec![0; max_order];
56            for order in 1..=max_order {
57                let possible_matches = translation_tokens.len().saturating_sub(order - 1);
58                if possible_matches > 0 {
59                    possible_matches_by_order[order - 1] += possible_matches
60                }
61            }
62
63            let translation_ngram_counts = get_token_ngram_counter(&translation_tokens, max_order);
64            let mut merged_ref_ngram_counts = AHashMap::new();
65            for reference_tokens in references_tokens.iter() {
66                let reference_ngram_counts = get_token_ngram_counter(reference_tokens, max_order);
67                for (key, value) in reference_ngram_counts {
68                    merged_ref_ngram_counts
69                        .entry(key)
70                        .and_modify(|v| *v = max(*v, value))
71                        .or_insert(value);
72                }
73            }
74
75            // overlap count
76            let mut matches_by_order = vec![0; max_order];
77            for (k, v) in translation_ngram_counts {
78                if merged_ref_ngram_counts.contains_key(k) {
79                    matches_by_order[k.len() - 1] += min(merged_ref_ngram_counts[k], v);
80                } else {
81                    continue;
82                }
83            }
84            Stat(
85                translation_length,
86                reference_length,
87                possible_matches_by_order,
88                matches_by_order,
89            )
90        })
91        .reduce_with(agg_stat);
92
93    let Stat(translation_length, reference_length, possible_matches_by_order, matches_by_order) =
94        match stat_result {
95            None => panic!("Pair statistics calculation got empty result"),
96            Some(stats) => stats,
97        };
98
99    // precisions calculation
100    let mut precisions: Vec<f64> = vec![0.0; max_order];
101    for i in 0..max_order {
102        match smooth {
103            true => {
104                precisions[i] = (matches_by_order[i] as f64 + 1.0)
105                    / (possible_matches_by_order[i] as f64 + 1.0);
106            }
107            false => {
108                if possible_matches_by_order[i] > 0 {
109                    precisions[i] =
110                        (matches_by_order[i] as f64) / (possible_matches_by_order[i] as f64)
111                } else {
112                    precisions[i] = 0.0;
113                }
114            }
115        }
116    }
117
118    let mut geo_mean = 0.0;
119
120    if precisions.iter().fold(f64::INFINITY, |a, &b| a.min(b)) > 0.0 {
121        let p_log_sum: f64 =
122            (1.0 / max_order as f64) * precisions.iter().map(|&x| x.ln()).sum::<f64>();
123        geo_mean = p_log_sum.exp();
124    }
125
126    let length_ratio: f64 = translation_length as f64 / reference_length as f64;
127    let mut brevity_penalty = 1.0;
128    if length_ratio <= 1.0 {
129        brevity_penalty = (1.0 - 1.0 / length_ratio).exp();
130    }
131    let bleu = geo_mean * brevity_penalty;
132
133    BleuScore {
134        bleu,
135        precisions,
136        brevity_penalty,
137        length_ratio,
138        translation_length,
139        reference_length,
140    }
141}

143fn add_vectors<T: Add<Output = T>>(vec1: Vec<T>, vec2: Vec<T>) -> Vec<T> {
144    // Ensure both vectors have the same length
145    assert_eq!(vec1.len(), vec2.len(), "Vectors must have the same length");
146    // Add elements of vec1 and vec2 element by element
147    let result: Vec<_> = vec1.into_iter().zip(vec2).map(|(a, b)| a + b).collect();
148    result
149}

151#[cfg(test)]
152mod test {
153    use crate::bleu::{add_vectors, agg_stat, compute_score, Stat};
154    #[test]
155    fn test_bleu() {
156        let references: Vec<Vec<String>> = vec![vec!["Hello, World!".to_string()]];
157        let predictions: Vec<String> = vec!["Yellow, World!".to_string()];
158        let max_order: usize = 4;
159        let smooth: bool = true;
160        let res = compute_score(&references, &predictions, max_order, smooth);
161        // (0.668740304976422, [0.8, 0.75, 0.6666666666666666, 0.5], 1.0, 1.0, 4, 4)
162        println!("result: {:?}", res);
163        assert!((res.bleu - 0.668740304976422).abs() < 1e-10);
164    }
165    #[test]
166    fn test_add_vectors() {
167        let v1 = vec![0, 1];
168        let v2 = vec![1, 2];
169        assert_eq!(vec![1, 3], add_vectors(v1, v2));
170    }
171
172    #[test]
173    fn test_stat_agg() {
174        let s1 = Stat(0, 1, vec![0, 1], vec![1, 2]);
175        let s2 = Stat(1, 2, vec![1, 2], vec![3, 4]);
176        assert_eq!(agg_stat(s1, s2), Stat(1, 3, vec![1, 3], vec![4, 6]));
177    }

179    #[test]
180    fn test_bleu_multi_ref() {
181        // Ref 1: "a"
182        // Ref 2: "a"
183        // Pred: "a a"
184        // Max ref count for "a" is 1.
185        // Pred count for "a" is 2.
186        // Clipped count is min(1, 2) = 1.
187        // Precision for 1-gram: 1/2 = 0.5.
188        let references: Vec<Vec<String>> = vec![vec!["a".to_string(), "a".to_string()]];
189        let predictions: Vec<String> = vec!["a a".to_string()];
190        let max_order: usize = 1;
191        let smooth: bool = false;
192        let res = compute_score(&references, &predictions, max_order, smooth);
193
194        // Precision should be 0.5
195        assert!((res.precisions[0] - 0.5).abs() < 1e-10);
196    }
